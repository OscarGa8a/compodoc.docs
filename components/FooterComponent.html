import { SharedElement } from './../../interfaces/shared-element.interface';
import { Component, OnInit, OnDestroy } from '@angular/core';

/**
 * Footer de la página
 */
@Component({
  selector: 'blog-footer',
  templateUrl: './footer.component.html',
  styleUrls: ['./footer.component.css']
})
export class FooterComponent implements OnInit, OnDestroy {

  /**
   * @ignore
   */
  currentYear: number = new Date().getFullYear();
  /**
   * Comentario inicial 1
   * Comentario inicial 2
   */
  author = 'Anartz Mugika Ledo';
  /**
   * Comentario inicial 1
   *
   * Comentario inicial 2
   */
  sharedElements: SharedElement[] = [
    {
      url: 'https://twitter.com/',
      path: 'mugan86',
      icon: 'fab fa-twitter'
    },
    {
      url: 'https://github.com/',
      path: 'mugan86',
      icon: 'fab fa-github'
    },
    {
      url: 'https://www.linkedin.com/in/',
      path: 'anartz-mugika-0007a062',
      icon: 'fab fa-linkedin'
    },
    {
      url: 'https://bintray.com/',
      path: 'amugika/maven',
      icon: 'fas fa-frog'
    },
    {
      url: 'https://npmjs.com/',
      path: '~mugan86',
      icon: 'fab fa-npm'
    }
  ];

  /**
   * Función que realiza un saludo
   *
   * {@link HeaderComponent}
   *
   * [Header]{@link HeaderComponent}
   *
   * {@link HeaderComponent|Header}
   *
   * [Header#bgUrl]{@link HeaderComponent#bgUrl}
   * @returns Devuelve un mensaje de bienvenida
   */
  bienvenida(): string {
    return 'Hola bienvenido a la documentacion';
  }

  /**
   * Función que realiza un saludo
   *
   * [Buscador de Google]{@link http://www.google.com}
   *
   * {@link http://www.apple.com|Pagina de Apple}
   *
   * {@link https://github.com Repositorio de GitHub}
   * @param nombre Nombre de la persona a la que se saluda
   * @param apellido Apellido de la persona a la que se saluda
   * @returns Devuelve un mensaje de bienvenida
   */
  bienvenidaPersonalizada(nombre: string, apellido: string): string {
    return `Hola ${nombre} ${apellido}, bienvenido a la documentacion`;
  }

  /**
   * Funcion que saluda al usuario
   * @example
   * Tenemos como nombre Oscar
   * Tenemos como apellido Garcia
   * Realiza el siguiente saludo \"hola Oscar Garcia\"
   */
  saludar(): string {
    return 'Hola';
  }

  /**
   * Bienvenidos
   */
  private holaMundo(): string {
    return 'Saludando';
  }

  /**
   * Bienvenidos2
   */
  protected holaMundo2(): string {
    return 'Saludando 2';
  }

  /**
   * OnInit
   */
  ngOnInit() {
    console.log('OnInit');
  }

  /**
   * OnDestroy
   */
  ngOnDestroy() {
    console.log('OnDestroy');
  }
}
